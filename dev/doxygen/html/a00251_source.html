99 SecondDerivative<ST> secondDerivativeImpl(const PrincipalStretches<ST>& lambda) const {
100 auto dS = SecondDerivative<ST>::Zero().eval();
101 const ScalarType J = Impl::determinantFromPrincipalValues(lambda);
102
103 auto lam = lambda.array();
104 dS = J / (lambda * lambda.transpose()).array();
105 dS.diagonal() = lam.square().inverse() * (lam.square().inverse() - J) + (3.0 / lam.pow(4));
106 dS.array() *= mu_;
107
108 return dS;
109 }
