28struct VonMises
29{
37 template <typename R>
38 double operator()(const R& resultArray, [[maybe_unused]] const auto& pos, [[maybe_unused]] const auto& fe,
39 [[maybe_unused]] const int comp) const {
40 auto sigma = fromVoigt(resultArray, false);
41
42 auto I2 = 1.0 / 2.0 * (sigma.squaredNorm() - 1.0 / 3.0 * Dune::power(sigma.trace(), 2));
43 return std::sqrt(3.0 * I2);
44 }
45
50 constexpr static std::string name() { return "VonMises"; }
51
56 constexpr static int ncomps() { return 1; }
57};

65struct HydrostaticStress
66{
74 template <typename R>
75 double operator()(const R& resultArray, [[maybe_unused]] const auto& pos, [[maybe_unused]] const auto& fe,
76 [[maybe_unused]] const int comp) const {
77 const auto sigma = fromVoigt(resultArray, false);

103struct PrincipalStress
104{
111 double operator()(const auto& resultArray, [[maybe_unused]] const auto& pos, [[maybe_unused]] const auto& fe,
112 const int comp) const {
113 auto mat = fromVoigt(resultArray, false);
114 Eigen::SelfAdjointEigenSolver<decltype(mat)> eigensolver(mat, Eigen::EigenvaluesOnly);
115 return eigensolver.eigenvalues().reverse()[comp];
116 }
117
122 constexpr static std::string name() { return "PrincipalStress"; }
123
128 constexpr static int ncomps() { return dim; }

137struct Triaxiality
138{
146 template <typename R>
147 double operator()(const R& resultArray, [[maybe_unused]] const auto& pos, [[maybe_unused]] const auto& fe,
148 [[maybe_unused]] const int comp) const {
149 auto sigeq = VonMises{}(resultArray, pos, fe, 0);

171struct PolarStress
172{
173 PolarStress(const Dune::FieldVector<double, 2>& origin)
174 : origin_(origin) {}
175
183 template <typename R>
184 double operator()(const R& resultArray, const auto& pos, const auto& fe, const int comp) const {
185 static_assert(R::CompileTimeTraits::RowsAtCompileTime == 3, "PolarStress is only valid for 2D.");
186 if (comp > 2)
187 DUNE_THROW(Dune::RangeError, "PolarStress: Comp out of range.");
188
189 // Offset to center the coordinate system in the reference geometry
190 Dune::FieldVector<double, 2> posGlobal = fe.geometry().global(pos) - origin_;
191 auto theta = std::atan2(posGlobal[1], posGlobal[0]);
192
193 const auto sigma = fromVoigt(resultArray, false);
194 Eigen::Rotation2D<double> r(theta);
195
196 // deliberately not evaluating this that it stays an expression for below
197 auto polarStress = r.inverse() * sigma * r;
198 switch (comp) {
199 case 0:
200 return polarStress(0, 0);
201 case 1:
202 return polarStress(1, 1);
203 case 2:
204 return polarStress(1, 0);
205 default:
206 __builtin_unreachable();
207 }
208 }
209
214 constexpr static std::string name() { return "PolarStress"; }
215
220 constexpr static int ncomps() { return 3; }
221
222private:
223 Dune::FieldVector<double, 2> origin_;
224};
