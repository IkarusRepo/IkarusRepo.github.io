91 void bindImpl() {
92 const auto& localView = underlying().localView();
93 const auto& element = localView.element();
94 auto& firstChild = localView.tree().child(0);
95 const auto& fe = firstChild.finiteElement();
96 geo_ = std::make_shared<const Geometry>(element.geometry());
97 numberOfNodes_ = fe.size();
98 order_ = 2 * (fe.localBasis().order());
99 localBasis_ = Dune::CachedLocalBasis(fe.localBasis());
100 if constexpr (requires { element.impl().getQuadratureRule(order_); })
101 if (element.impl().isTrimmed())
102 localBasis_.bind(element.impl().getQuadratureRule(order_), Dune::bindDerivatives(0, 1));
103 else
104 localBasis_.bind(Dune::QuadratureRules<double, myDim>::rule(element.type(), order_),
105 Dune::bindDerivatives(0, 1));
106 else
107 localBasis_.bind(Dune::QuadratureRules<double, myDim>::rule(element.type(), order_), Dune::bindDerivatives(0, 1));
108 }

154 auto materialTangent(const Eigen::Vector<ScalarType, strainDim>& strain) const {

214 template <template <typename, int, int> class RT>
215 requires(supportsResultType<RT>())
216 auto calculateAtImpl(const Requirement& req, const Dune::FieldVector<double, Traits::mydim>& local,
217 Dune::PriorityTag<1>) const {
218 using namespace Dune::DerivativeDirections;
219
220 using RTWrapper = ResultWrapper<RT<typename Traits::ctype, myDim, Traits::worlddim>, ResultShape::Vector>;
221 if constexpr (isSameResultType<RT, ResultTypes::PK2Stress>) {
222 const auto uFunction = displacementFunction(req);
223 const auto H = uFunction.evaluateDerivative(local, Dune::wrt(spatialAll), Dune::on(gridElement));
224 const auto E = (0.5 * (H.transpose() + H + H.transpose() * H)).eval();
225
226 return RTWrapper{mat_.template stresses<StrainTags::greenLagrangian>(toVoigt(E))};
227 }
228 }

248 template <typename ScalarType>
249 void calculateMatrixImpl(
250 const Requirement& par, const MatrixAffordance& affordance, typename Traits::template MatrixType<> K,
251 const std::optional<std::reference_wrapper<const Eigen::VectorX<ScalarType>>>& dx = std::nullopt) const {
252 using namespace Dune::DerivativeDirections;
253 using namespace Dune;
254 const auto uFunction = displacementFunction(par, dx);
255 const auto eps = strainFunction(par, dx);
256 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
257 const double intElement = geo_->integrationElement(gp.position()) * gp.weight();
258 const auto EVoigt = (eps.evaluate(gpIndex, on(gridElement))).eval();
259 const auto u = (uFunction.evaluate(gpIndex, on(gridElement))).eval();
260 const auto C = materialTangent(EVoigt);
261
262 const auto stresses = getStress(EVoigt);
263 for (size_t i = 0; i < numberOfNodes_; ++i) {
264 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
265 for (size_t j = 0; j < numberOfNodes_; ++j) {
266 const auto bopJ = eps.evaluateDerivative(gpIndex, wrt(coeff(j)), on(gridElement));
267 const auto kgIJ = eps.evaluateDerivative(gpIndex, wrt(coeff(i, j)), along(stresses), on(gridElement));
268 K.template block<myDim, myDim>(i * myDim, j * myDim) += (bopI.transpose() * C * bopJ + kgIJ) * intElement;
269 }
270 }
271 }
272 }

274 template <typename ScalarType>
275 auto calculateScalarImpl(const Requirement& par, ScalarAffordance affordance,
276 const std::optional<std::reference_wrapper<const Eigen::VectorX<ScalarType>>>& dx =
277 std::nullopt) const -> ScalarType {
278 using namespace Dune::DerivativeDirections;
279 using namespace Dune;
280
281 const auto eps = strainFunction(par, dx);
282 const auto& lambda = par.parameter();
283 ScalarType energy = 0.0;
284
285 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
286 const auto EVoigt = (eps.evaluate(gpIndex, on(gridElement))).eval();
287 const auto internalEnergy = getInternalEnergy(EVoigt);
288 energy += internalEnergy * geo_->integrationElement(gp.position()) * gp.weight();
289 }
290
291 return energy;
292 }

294 template <typename ScalarType>
295 void calculateVectorImpl(
296 const Requirement& par, VectorAffordance affordance, typename Traits::template VectorType<ScalarType> force,
297 const std::optional<std::reference_wrapper<const Eigen::VectorX<ScalarType>>>& dx = std::nullopt) const {
298 using namespace Dune::DerivativeDirections;
299 using namespace Dune;
300 const auto eps = strainFunction(par, dx);
301
302 // Internal forces
303 for (const auto& [gpIndex, gp] : eps.viewOverIntegrationPoints()) {
304 const double intElement = geo_->integrationElement(gp.position()) * gp.weight();
305 const auto EVoigt = (eps.evaluate(gpIndex, on(gridElement))).eval();
306 const auto stresses = getStress(EVoigt);
307 for (size_t i = 0; i < numberOfNodes_; ++i) {
308 const auto bopI = eps.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
309 force.template segment<myDim>(myDim * i) += bopI.transpose() * stresses * intElement;
310 }
311 }
312 }
