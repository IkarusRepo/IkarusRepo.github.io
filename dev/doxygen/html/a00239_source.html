148 template <typename Derived>
149 void initUnknownStrains(Eigen::MatrixBase<Derived>& E) const {
150 for (size_t i = 0; i < fixedPairs.size(); ++i) {
151 ScalarType initialVal = E(fixedPairs[i].row, fixedPairs[i].col);
152 if constexpr (strainTag == StrainTags::deformationGradient or strainTag == StrainTags::rightCauchyGreenTensor) {
153 if (Dune::FloatCmp::eq(initialVal, ScalarType(0.0)) and (fixedPairs[i].row == fixedPairs[i].col))
154 initialVal = ScalarType(1.0);
155 }
156 if (fixedPairs[i].row != fixedPairs[i].col)
157 initialVal = ScalarType(0.0);
158 E(fixedPairs[i].row, fixedPairs[i].col) = E(fixedPairs[i].col, fixedPairs[i].row) = initialVal;
159 }
160 }

168 template <typename Derived>
169 auto reduceStress(const Eigen::MatrixBase<Derived>& Eraw) const {
170 auto E = Impl::maybeFromVoigt(Eraw);
171 initUnknownStrains(E);
172
173 std::array<size_t, fixedDiagonalVoigtIndicesSize> fixedDiagonalVoigtIndices;
174 for (size_t ri = 0; auto i : fixedVoigtIndices) {
175 auto indexPair = fromVoigt(i);
176 if (indexPair[0] == indexPair[1])
177 fixedDiagonalVoigtIndices[ri++] = i;
178 }
179
180 auto f = [&](auto&) {
181 auto S = matImpl_.template stresses<Underlying::strainTag, true>(E);
182 return S(fixedDiagonalVoigtIndices).eval();
183 };
184 auto df = [&](auto&) {
185 auto moduli = (matImpl_.template tangentModuli<Underlying::strainTag, true>(E)).eval();
186 return (moduli(fixedDiagonalVoigtIndices, fixedDiagonalVoigtIndices) / Underlying::derivativeFactor).eval();
187 };
188
189 auto Er = E(fixedDiagonalVoigtIndices, fixedDiagonalVoigtIndices).eval().template cast<ScalarType>();
190 auto nonOp = Ikarus::NonLinearOperator(functions(f, df), parameter(Er));
191
192 auto linearSolver = [](auto& r, auto& A) { return (A.inverse() * r).eval(); };
193 auto updateFunction = [&](auto& /* Ex33 */, auto& ecomps) {
194 for (int ri = 0; auto i : fixedDiagonalVoigtIndices) {
195 auto indexPair = fromVoigt(i);
196 E(indexPair[0], indexPair[1]) += ecomps(ri++);
197 }
198 };
199
200 int minIter = isAutoDiff ? 1 : 0;
201 // THE CTAD is broken for designated initializers in clang 16, when we drop support this can be simplified
202 NewtonRaphsonConfig<decltype(linearSolver), decltype(updateFunction)> nrs{
203 .parameters = {.tol = tol_, .maxIter = 100, .minIter = minIter},
204 .linearSolver = linearSolver,
205 .updateFunction = updateFunction
206 };
207
208 auto nr = createNonlinearSolver(std::move(nrs), nonOp);
209 if (!static_cast<bool>(nr->solve()))
210 DUNE_THROW(Dune::MathError, "The stress reduction of material " << nameImpl() << " was unsuccessful\n"
211 << "The strains are\n"
212 << E << "\n The stresses are\n"
213 << f(Er));
214 return std::make_pair(nonOp, E);
215 }
216
217 Underlying matImpl_;
218 double tol_{};
219};
