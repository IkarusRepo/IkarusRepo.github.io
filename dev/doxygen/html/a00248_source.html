101 template <template <typename, int, int> class RT>
102 auto calculateAtImpl(const Requirement& req, const Dune::FieldVector<double, Traits::mydim>& local,
103 Dune::PriorityTag<2>) const {
104 if (isDisplacementBased())
105 return underlying().template calculateAtImpl<RT>(req, local, Dune::PriorityTag<1>());
106
107 using namespace Dune::Indices;
108 using namespace Dune::DerivativeDirections;
109 auto resultVector = underlying().template calculateAtImpl<RT>(req, local, Dune::PriorityTag<1>());
110 if constexpr (isSameResultType<RT, ResultTypes::linearStress>) {
111 using RTWrapper = ResultWrapper<RT<typename Traits::ctype, Traits::mydim, Traits::worlddim>, ResultShape::Vector>;
112 RTWrapper resultWrapper;
113
114 auto calculateAtContribution = [&]<typename EAST>(const EAST& easFunction) {
115 typename EAST::DType D;
116 calculateDAndLMatrix(easFunction, req, D, L_);
117 const auto ufunc = underlying().displacementFunction(req);
118 const auto disp = Dune::viewAsFlatEigenVector(ufunc.coefficientsRef());
119 const auto C = underlying().materialTangentFunction(req);
120 const auto alpha = (-D.inverse() * L_ * disp).eval();
121 const auto M = easFunction.calcM(local);
122 const auto CEval = C(local);
123 auto easStress = (CEval * M * alpha).eval();
124 resultWrapper = resultVector.asVec() + easStress;
125 };
126 easVariant_(calculateAtContribution);
127 return resultWrapper;
128 }
129 }

157 template <typename ScalarType>
158 void calculateMatrixImpl(
159 const Requirement& par, const MatrixAffordance& affordance, typename Traits::template MatrixType<> K,
160 const std::optional<std::reference_wrapper<const Eigen::VectorX<ScalarType>>>& dx = std::nullopt) const {
161 using namespace Dune::DerivativeDirections;
162 using namespace Dune;
163 easApplicabilityCheck();
164 if (isDisplacementBased())
165 return;
166
167 decltype(auto) LMat = [this]() -> decltype(auto) {
168 if constexpr (std::is_same_v<ScalarType, double>)
169 return [this]() -> Eigen::MatrixXd& { return L_; }();
170 else
171 return Eigen::MatrixX<ScalarType>{};
172 }();
173
174 auto calculateMatrixContribution = [&]<typename EAST>(const EAST& easFunction) {
175 typename EAST::DType D;
176 calculateDAndLMatrix(easFunction, par, D, LMat, dx);
177
178 K.template triangularView<Eigen::Upper>() -= LMat.transpose() * D.inverse() * LMat;
179 K.template triangularView<Eigen::StrictlyLower>() = K.transpose();
180 };
181 easVariant_(calculateMatrixContribution);
182 }

195 template <typename ScalarType>
196 void calculateVectorImpl(
197 const Requirement& par, VectorAffordance affordance, typename Traits::template VectorType<ScalarType> force,
198 const std::optional<std::reference_wrapper<const Eigen::VectorX<ScalarType>>>& dx = std::nullopt) const {
199 easApplicabilityCheck();
200 using namespace Dune;
201 using namespace Dune::DerivativeDirections;
202 const auto uFunction = underlying().displacementFunction(par, dx);
203 auto strainFunction = underlying().strainFunction(par, dx);
204 const auto& numberOfNodes = underlying().numberOfNodes();
205
206 auto calculateForceContribution = [&]<typename EAST>(const EAST& easFunction) {
207 typename EAST::DType D;
208 calculateDAndLMatrix(easFunction, par, D, L_);
209
210 decltype(auto) LMat = [this]() -> decltype(auto) {
211 if constexpr (std::is_same_v<ScalarType, double>)
212 return [this]() -> Eigen::MatrixXd& { return L_; }();
213 else
214 return Eigen::MatrixX<ScalarType>{};
215 }();
216 const auto disp = Dune::viewAsFlatEigenVector(uFunction.coefficientsRef());
217 const auto alpha = (-D.inverse() * L_ * disp).eval();
218 const auto geo = underlying().localView().element().geometry();
219 auto C = underlying().materialTangentFunction(par);
220
221 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
222 const auto M = easFunction.calcM(gp.position());
223 const double intElement = geo.integrationElement(gp.position()) * gp.weight();
224 const auto CEval = C(gpIndex);
225 auto stresses = (CEval * M * alpha).eval();
226 for (size_t i = 0; i < numberOfNodes; ++i) {
227 const auto bopI = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
228 force.template segment<Traits::worlddim>(Traits::worlddim * i) += bopI.transpose() * stresses * intElement;
229 }
230 }
231 };
232 easVariant_(calculateForceContribution);
233 }

243 template <typename ScalarType, int enhancedStrainSize>
244 void calculateDAndLMatrix(
245 const auto& easFunction, const auto& par, Eigen::Matrix<double, enhancedStrainSize, enhancedStrainSize>& DMat,
246 Eigen::MatrixX<ScalarType>& LMat,
247 const std::optional<std::reference_wrapper<const Eigen::VectorX<ScalarType>>>& dx = std::nullopt) const {
248 using namespace Dune;
249 using namespace Dune::DerivativeDirections;
250
251 auto strainFunction = underlying().strainFunction(par, dx);
252 const auto C = underlying().materialTangentFunction(par);
253 const auto geo = underlying().localView().element().geometry();
254 const auto numberOfNodes = underlying().numberOfNodes();
255 DMat.setZero();
256 LMat.setZero(enhancedStrainSize, underlying().localView().size());
257 for (const auto& [gpIndex, gp] : strainFunction.viewOverIntegrationPoints()) {
258 const auto M = easFunction.calcM(gp.position());
259 const auto CEval = C(gpIndex);
260 const double detJTimesW = geo.integrationElement(gp.position()) * gp.weight();
261 DMat += M.transpose() * CEval * M * detJTimesW;
262 for (size_t i = 0U; i < numberOfNodes; ++i) {
263 const size_t I = Traits::worlddim * i;
264 const auto Bi = strainFunction.evaluateDerivative(gpIndex, wrt(coeff(i)), on(gridElement));
265 LMat.template block<enhancedStrainSize, Traits::worlddim>(0, I) += M.transpose() * CEval * Bi * detJTimesW;
266 }
267 }
268 }
