45 // Since base classes are initialized, in declaration order before member variables, we have to make sure the
46 // localview_ object of FE is defined. To do this we add this artificial inheritance by inheriting below first from
47 // FEInit, which initializes the localview object first, siche we the other FEMixin base class constructors are called
48 template <typename PreFE, typename FE>
49 struct FEInit
50 {
51 using Traits = PreFE::Traits;
52 using BasisHandler = Traits::BasisHandler;
53 using LocalView = typename Traits::LocalView;
54 friend FE;
55 FEInit(const BasisHandler& basisHandler)
56 : localView_{[&]() -> LocalView {
57 if constexpr (Traits::useFlatBasis)
58 return basisHandler.flat().localView();
59 else
60 return basisHandler.untouched().localView();
61 }()} {}
62
63 private:
64 LocalView localView_;
65 };
66} // namespace Impl
